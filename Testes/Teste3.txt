class Teste3

int i;
int soma;
float media; // corrigido: operação de divisão retorna sempre float
int qtd; // corrigido: erro 'qtd' não declarado
int altura; // corrigido erro 'altura' não declarado

init
    soma = 0;
    write("Quantos dados deseja informar?");
    read(qtd);
    if (qtd >= 2){
        i=0;
        do{
            write("Altura: ");
            read (altura);
            soma = soma+altura;
            i = i + 1;
        }while( i < qtd);
        media = soma / qtd;
        write("Media: ");
        write (media);
    }
    else{
        write("Quantidade inválida.");
    };
stop
