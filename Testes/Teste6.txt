/** Verificando fluxo de controle
Programa com while aninhados **/

int i, j, valor, qtd;

init
    valor = 0;
    write("Quantos vezes (ao quadrado) deseja rodar?" );
    read(qtd);
    i = 0;

    while(i < qtd){
        j = 0;
        while(j < qtd) {
            valor = j * 5;
            j = j + 1;
        }

        i = i + 1;
    }

    write("O calculo resultou em: ");
    write(valor);
stop
